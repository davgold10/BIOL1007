# heightPlot <- qplot(x=starwars2$species, y=starwars2$height, geom="boxplot", fill="green")

heightPlot <- ggplot(data=starwars2) +
  aes(x=species, y=height) +
  geom_boxplot(fill=c("red", "orange", "yellow", "green", "blue", "darkblue", "purple", "magenta", "black")) +
  xlab("Species") +
  ylab("Height (cm)") +
  theme_dark() +
  annotate("text", x=9, y=max(starwars2$height), label="p=1.2E-0.5")
